#hash

#variações:

#não destrói código
#colocar arquivos no index/staging
--soft
git reset --soft HEAD~1

#não destrói código
#reverte para work directory, antes do index/staging
#default
--mixed
git reset --mixed HEAD~1

#destrói código/arquivos
#não move para outros estados
--hard
git reset --hard HEAD~1



#head vai apontar para o hash
git reset <hash number>

#move a HEAD 1 para trás
git reset HEAD~1


#comandos para reset
git reset <hash>
git commit -m "message"
git pull origin master
#resolver conflitos
git push origin master





#orientado a HEAD ou hash
#reverte para todas as alterações do commit escolhido
git revert

#gera um novo commit, revertendo o commit HEAD ou hash
#se adicionar um arquivo, será removido
#código será revertido para o commit escolhido

git revert HEAD~1
git revert <hash number>

#comandos para revert
git revert <hash>
#cria um novo commit
git push origin master
